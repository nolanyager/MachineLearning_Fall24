from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler
import random

# Function to approximate the data matrix X using the first k principal components
def pca_approx(X, k):
    U, S, Vt = np.linalg.svd(X, full_matrices=False)
    Uk = U[:, :k]
    Sk = np.diag(S[:k])
    Vtk = Vt[:k, :]
    return Uk @ Sk @ Vtk

# Function to compute the cross-validated error of approximating X using the first k principal components
def pca_cv_error(X, k, folds):
    unique_folds = np.unique(folds)
    means = X.mean(axis=0)
    errs = np.zeros(len(unique_folds))

    for idx, f in enumerate(unique_folds):
        X_cv = X.copy()

        # Replace fold f values with column means
        for i in range(X.shape[1]):
            for j in range(X.shape[0]):
                if folds[j, i] == f:
                    X_cv[j, i] = means[i]

        # Approximate X_cv using the first k principal components
        X_approx = pca_approx(X_cv, k)

        # Calculate the error only for replaced values
        err = 0.0
        for i in range(X.shape[1]):
            for j in range(X.shape[0]):
                if folds[j, i] == f:
                    err += (X_approx[j, i] - X[j, i]) ** 2
        errs[idx] = err

    return np.sum(errs) / X.size
